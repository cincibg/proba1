import { networkInterfaces } from 'os';

function getMacAddresses() {
  const interfaces = networkInterfaces();
  const macAddresses = [];
  
  for (const [name, nets] of Object.entries(interfaces)) {
    if (nets) {
      for (const net of nets) {
        if (net.mac && net.mac !== '00:00:00:00:00:00') {
          macAddresses.push({
            interface: name,
            mac: net.mac,
            family: net.family,
            address: net.address
          });
        }
      }
    }
  }
  
  return macAddresses;
}
